Exclude samples with missing labels#

Show code cell source Hide code cell source 
# drop the samples with missing labels for the ELN AML 2022 Diagnosis
df = df[~df['ELN AML 2022 Diagnosis'].isna()]

# Standardize the `df[['PaCMAP 1', 'PaCMAP 2', 'PaCMAP 3', 'PaCMAP 4', 'PaCMAP 5']]` so mean = 0 and std = 1
X = df[['PaCMAP 1', 'PaCMAP 2', 'PaCMAP 3', 'PaCMAP 4', 'PaCMAP 5']]
X = (X - X.mean()) / X.std()

# join the standardized PaCMAP coordinates with the labels
df = pd.concat([X, df[['ELN AML 2022 Diagnosis', 'Train Test']]], axis=1)

# Save the data
df.to_csv(output_path+'pacmap_output/pacmap_5d_output_acute_leukemia_cleaned.csv')

SPPL#

  • SPPL: Sum-Product Probabilistic Language

  • Github: probsys/sppl

  • Paper: SPPL: Probabilistic Programming with Fast Exact Symbolic Inference

  • Intro on SPNs: Visualizing and understanding Sum-Product Networks

%load_ext sppl.magics

%%sppl model

# Population model.
sex ~= choice({'female': .3307, 'male': .6693})
if (sex == 'female'):
