Nanopore Dataset#

Basecall and align using Dorado and minimap2#

dorado basecaller \
sup,5mCG_5hmCG \
/path/to/pod5/ \
--reference ref/hg38.fna \
--trim adapters \
--min-qscore 15 > {$SAMPLE}_sup5_unsorted.bam

Sort and index BAM file#

samtools sort \
--threads 8 \
--write-index \
-m 8G \
-o "{$SAMPLE}.bam" \
-O BAM \
--reference ref/hg38.fna \
"{$SAMPLE}"

Create a BedMethyl file with Modkit#

modkit pileup \
"$bam_file" \
"$bed_file" \
-t 32 \
--combine-strands \
--cpg \
--ignore h \
--ref ref/hg38.fna \
--no-filtering

Get diagnosis and prognosis with ALMA-classifier#

alma-classifier \
--input "$bed_file" \
--output "{$SAMPLE}_predictions.xlsx"
